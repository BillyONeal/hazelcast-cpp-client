75  void readData(Input &dataInput) {
76  type = dataInput.readInt();
77  int classId = dataInput.readInt();
78 
79  if (classId != NO_CLASS_ID) {
80  int factoryId = dataInput.readInt();
81  int version = dataInput.readInt();
82 
83  int classDefSize = dataInput.readInt();
84  SerializationContext *serializationContext = dataInput.getSerializationContext();
85  if (serializationContext->isClassDefinitionExists(factoryId, classId, version)) {
86  cd = serializationContext->lookup(factoryId, classId, version);
87  dataInput.skipBytes(classDefSize);
88  } else {
89  std::auto_ptr< std::vector<byte> > classDefBytes (new std::vector<byte> (classDefSize));
90  dataInput.readFully(*(classDefBytes.get()));
91  cd = serializationContext->createClassDefinition(factoryId, classDefBytes);
92  }
93  }
94  int size = dataInput.readInt();
95  if (size > 0) {
96  this->buffer->resize(size);
97  dataInput.readFully(*(buffer.get()));
98  }
99  partitionHash = dataInput.readInt();
100  }
